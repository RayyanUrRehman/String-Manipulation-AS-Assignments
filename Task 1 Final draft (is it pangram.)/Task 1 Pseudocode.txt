Task1: Find if a string is Pangram

PSEUDOCODE:


DECLARE Str1, alpha : STRING
DECLARE isPangram : BOOLEAN 
DECLARE i : INTEGER
DECLARE Char1 : CHARACTER


Str1 ← ""
alphas ← "abcdefghijklmnopqrstuvwxyz"
isPangram ← TRUE
i ← 0
Char1 ← ''

INPUT "Enter string to process: ", Str1
Str1 ← LCase(Str1)

FOR i ← 1 TO LEN(alphas)
  Char1 ← MID(alphas, i, 1)
  IF LOCATE(Str1, Char1) = 0 THEN isPangram ← FALSE
  IF isPangram = FALSE THEN Exit FOR
NEXT 

IF isPangram = TRUE
  THEN
    OUTPUT "Entered string is a Pangram."
  ELSE
    OUTPUT "Entered string is NOT a Pangram."
END IF
