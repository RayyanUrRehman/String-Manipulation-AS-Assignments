Task 3: Count and output the number of an entered character in a string.
Also output seperate counts for alphabets (case neutral), digits, and other characters in same entered string.

Visual basic:

Module Module1

    Sub Main()
        'Declaration
        Dim str1 As String
        Dim char1 As Char
        Dim c As Char
        Dim totalalphas As Integer
        Dim totaldigits As Integer
        Dim totalchar1 As Integer
        Dim totalspchar As Integer
        Dim i As Integer

        'Initialization
        str1 = ""
        char1 = ""
        c = ""
        totalalphas = 0
        totalchar1 = 0
        totaldigits = 0
        totalspchar = 0
        i = 0

        'Input + Case neutralization
        Console.Write("Enter string to process: ")
        str1 = Console.ReadLine
        Console.Write("Enter character to count: ")
        char1 = Console.ReadLine
        str1 = LCase(str1)
        char1 = LCase(char1)

        'Process
        For i = 1 To Len(str1)
            c = Mid(str1, i, 1)
            If Asc(c) = Asc(char1) Then
                totalchar1 = totalchar1 + 1
                totalalphas = totalalphas + 1
            ElseIf Asc(c) >= 97 And Asc(c) <= 122 Then
                totalalphas = totalalphas + 1
            ElseIf Asc(c) >= 48 And Asc(c) <= 57 Then
                totaldigits = totaldigits + 1
            Else : totalspchar = totalspchar + 1
            End If
        Next

        'output
        Console.WriteLine(char1 & " has been written " & totalchar1 & " times in the string.")
        Console.WriteLine("Total number of alphabets entered are: " & totalalphas)
        Console.WriteLine("Total number of digits entered are: " & totaldigits)
        Console.WriteLine("Total number of special characters entered (including space) are: " & totalspchar)

        Console.ReadKey()




    End Sub

End Module
